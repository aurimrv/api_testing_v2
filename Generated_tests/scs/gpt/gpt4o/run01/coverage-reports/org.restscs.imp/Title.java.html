package org.restscs.imp;

public class Title
{
  
  public static String  subject(String sex, String title)
  {
  //CHECK PERSONAL TITLE CONSISTENT WITH SEX
    sex = sex.toLowerCase();
	title = title.toLowerCase();
    int result  = -1;
		if ("male".equals(sex)) {
			if ("mr".equals(title) || 
					"dr".equals(title) ||
					"sir".equals(title) ||
					"rev".equals(title) ||
					"rthon".equals(title) ||
					"prof".equals(title)) {
        result = 1;
      }
	  }
		else if ("female".equals(sex)) {
			if ("mrs".equals(title) || 
					"miss".equals(title) ||
					"ms".equals(title) ||
					"dr".equals(title) ||
					"lady".equals(title) ||
					"rev".equals(title) ||
					"rthon".equals(title) ||
					"prof".equals(title)){
        result = 0;
      }
	  }
		else if ("none".equals(sex)) {
			if ("dr".equals(title) ||
					"rev".equals(title) ||
					"rthon".equals(title) ||
					"prof".equals(title)){
        result = 2;
      }
	  }
    return "" + result;
  }
}
